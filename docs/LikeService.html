def self.like(user_id, likeable)
  like = get_like(user_id, likeable)

  if like.nil?
    likeable.likes.create!(user_id:)
    return { json: 'Object created', status: :ok }
  end
  like.destroy
  { json: 'Object destroyed', status: :ok }
end
